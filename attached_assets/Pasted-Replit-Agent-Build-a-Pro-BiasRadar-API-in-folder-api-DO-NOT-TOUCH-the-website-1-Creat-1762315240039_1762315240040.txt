     datasource db { provider = "sqlite" url = env("DATABASE_URL") }
     model Organization {
       id            String   @id @default(uuid())
       name          String
       email         String   @unique
       is_paid       Boolean  @default(false)
       api_key       String   @unique
       requests_made Int      @default(0)
       last_reset    DateTime @default(now())
